# 安装virtualenv环境
virtualenv env

# 下载源代码
git clone git@github.com:zhuguangjun2002/mymap.git

# 激活环境
source env/bin/activate

# 安装依赖
pip install -r mymap/requirements.txt

# 准备项目环境
cp project-secret/env-ubuntu mymap/.env
或者
cp project-secret/env-centos mymap/.env

# 执行`迁移`操作
python manage.py makemigrations
python manage.py  migrate


# 进入项目，运行开放服务器：
cd mymap/
./run-server.sh
